_get_RPK PROC
    push ebp
    mov ebp, esp
    push ebx
    push esi
    push edi
    
    sub esp, 15
    mov edi, esp
    xor eax, eax
    mov ecx, 15
    rep stosb
    mov edi, esp  ; reset edi

    mov esi, [ebp+8]      
    add esi, 6
    
    mov ecx, 4  

process_group:
    push ecx
    
    mov ecx, 5
process_char:
    mov al, [esi]
    inc esi
    
    call char_to_value
    
    push esi
    push edi
    push edi  ; in
    push edi  ; out
    call _mul_24@8
    pop edi
    pop esi
    
    movzx eax, al
    add [edi], al
    
    mov ebx, 1  ; Start from [edi+1]
carry_loop:
    adc byte ptr [edi+ebx], 0
    inc ebx
    cmp ebx, 14  ; Process bytes 1-14 (15 total)
    jbe carry_loop
    
    loop process_char
    
    pop ecx
    cmp ecx, 1
    je no_hyphen
    inc esi  ; skip '-'
no_hyphen:
    loop process_group
    
    mov eax, [edi+8]  
    bswap eax             


    add esp, 15
    pop edi
    pop esi
    pop ebx
    pop ebp
    ret
_get_RPK ENDP
