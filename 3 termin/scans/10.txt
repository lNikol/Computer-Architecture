.model flat
public _main

.code
_main PROC
    ; Inicjalizacja (2 MOV)
    mov eax, 42         ; 1. MOV
    mov ebx, 73         ; 2. MOV
    
    ; Stos (1 PUSH, 3 POP)
    push eax            ; 1. PUSH (druga cyfra=0, ale musi być min 1)
    pop ecx             ; 1. POP
    pop edx             ; 2. POP  
    pop esi             ; 3. POP (trzecia cyfra=1, ale specjalnie więcej dla demonstracji)
    
    ; Operacje arytmetyczne (0 ADD, 2 SUB)
    sub ebx, eax        ; 1. SUB
    sub ecx, 100        ; 2. SUB
    
    ; Mnożenie i dzielenie
    mul ebx             ; Mnożenie EAX*EBX
    cdq
    mov ecx, 5
    div ecx             ; Dzielenie EDX:EAX/ECX
    
    ; Skoki warunkowe
    cmp eax, 10
    jg wieksza         ; JMP warunkowe
    jmp mniejsza       ; JMP bezwarunkowe
    
wieksza:
    mov edi, 1         ; 3. MOV
    jmp kontynuuj
    
mniejsza:
    mov edi, 0         ; 4. MOV
    
kontynuuj:
    ; Losowe operacje
    xor esi, esi
    or edx, 0FFh
    
    ; Kończenie programu
    mov eax, 0         ; 5. MOV
    ret
_main ENDP
END
