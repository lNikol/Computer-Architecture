public _sortuj

.code
_sortuj PROC
    push ebp
    mov ebp, esp
    push ebx
    push esi
    push edi

    mov ebx, [ebp+8]
    mov ecx, [ebp+12]

    cmp ecx, 1
    jle koniec

    mov esi, 1

outer_loop:
    mov ecx, [ebx + esi*8]
    mov edx, [ebx + esi*8 + 4]

    mov edi, esi
    dec edi

inner_loop:
    cmp edi, 0
    jl insert

    mov eax, [ebx + edi*8 + 4]
    cmp eax, edx
    jg shift
    jl insert

    mov eax, [ebx + edi*8]
    cmp eax, ecx
    jbe insert

shift:
    mov eax, [ebx + edi*8]
    mov [ebx + edi*8 + 8], eax
    mov eax, [ebx + edi*8 + 4]
    mov [ebx + edi*8 + 12], eax

    dec edi
    jmp inner_loop

insert:
    mov [ebx + edi*8 + 8], ecx
    mov [ebx + edi*8 + 12], edx

    inc esi
    cmp esi, [ebp+12]
    jl outer_loop

koniec:
    pop edi
    pop esi
    pop ebx
    pop ebp
    ret
_sortuj ENDP

END
